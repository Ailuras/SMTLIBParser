134 class SHA256 {
135 public:
136 // Static function to get the SHA256 hash of a string
137 static std::string hash(const std::string &input) {
138 static SHA256 sha256; // A static SHA256 object
139 sha256.reset(); // Reset the state before using it
140 return sha256.process(input);
141 }
142
143 private:
144 uint32_t state[8]; // Hash values
145 uint64_t bitLength; // Total bit length of input
146
147 // SHA-256 constants (first 32 bits of the fractional parts of the square roots of the first 64 primes)
148 static constexpr uint32_t k[64] = {
149 0x428a2f98, 0x71374491, 0xb5c0fbcf, 0xe9b5dba5, 0x3956c25b, 0x59f111f1, 0x923f82a4, 0xab1c5ed5,
150 0xd807aa98, 0x12835b01, 0x243185be, 0x550c7dc3, 0x72be5d74, 0x80deb1fe, 0x9bdc06a7, 0xc19bf174,
151 0xe49b69c1, 0x3192c4fd, 0x2b8e1e4d, 0x7f2a7c67, 0x94f82a4d, 0x37533f4f, 0x5f8a1e10, 0x43172f60,
152 0x8e44a3f8, 0x88d7e7ed, 0xd0c8c12d, 0x35da6c31, 0x6e37a0df, 0x6071b9b1, 0xa62e53bc, 0x37b156f3,
153 0x8ff8ed2b, 0xa6e472e2, 0xa3c68864, 0x3be12e2e, 0x9d6c71fe, 0x3f775f6b, 0xa852adf7, 0x5b2266a7,
154 0xe1be9d35, 0x3da7096d, 0xf8c7032a, 0x2b4b9c0c, 0x438156fd, 0x2d35b6b9, 0x3ff58e3f, 0x94a15ec5,
155 0x4d27c0ee, 0x5a15a079, 0x771b6d91, 0x437f394f, 0x69e2a63f, 0x711c312f, 0x5b9d674f, 0x8e88c877
156 };
157
158 // Constructor, initializing the hash values (state)
159 SHA256() {
160 reset();
161 }
162
163 // Reset the state (called before each new hash calculation)
164 void reset() {
165 // Initial hash values (Constants)
166 state[0] = 0x6a09e667;
167 state[1] = 0xbb67ae85;
168 state[2] = 0x3c6ef372;
169 state[3] = 0xa54ff53a;
170 state[4] = 0x510e527f;
171 state[5] = 0x9b05688c;
172 state[6] = 0x1f83d9ab;
173 state[7] = 0x5be0cd19;
174
175 // Padding length and bit length initialization
176 bitLength = 0;
177 }
178
179 // Function to process the input string and return the hash
180 std::string process(const std::string &input) {
181 // Handle empty input
182 if (input.empty()) {
183 // Process a single block with just padding
184 std::vector<uint8_t> data(64, 0);
185 data[0] = 0x80; // Append 1 bit
186 data[63] = 0; // Message length is 0
187 processBlock(data.data());
188 return toHexString();
189 }
190
191 // Step 1: Padding the input
192 std::vector<uint8_t> data = padInput(input);
193
194 // Step 2: Processing each 512-bit block
195 for (size_t i = 0; i < data.size(); i += 64) {
196 processBlock(&data[i]);
197 }
198
199 // Step 3: Output the final hash
200 return toHexString();
201 }
202
203 // Padding the input according to the SHA-256 specification
204 std::vector<uint8_t> padInput(const std::string &input) {
205 std::vector<uint8_t> data(input.begin(), input.end());
206 bitLength = input.size() * 8;
207
208 // Append a 1 bit (0x80) to the input
209 data.emplace_back(0x80);
210
211 // Append 0 bits until the length is congruent to 448 mod 512
212 while ((data.size() * 8) % 512 != 448) {
213 data.emplace_back(0x00);
214 }
215
216 // Append the length of the original message (in bits) as a 64-bit big-endian integer
217 for (int i = 7; i >= 0; --i) {
218 data.emplace_back(static_cast<uint8_t>((bitLength >> (i * 8)) & 0xFF));
219 }
220
221 return data;
222 }
223
224 // Process a 512-bit block of data (64 bytes)
225 void processBlock(const uint8_t *block) {
226 uint32_t w[64];
227
228 // Break the block into 16 32-bit words
229 for (int i = 0; i < 16; i++) {
230 w[i] = (block[i * 4] << 24) |
231 (block[i * 4 + 1] << 16) |
232 (block[i * 4 + 2] << 8) |
233 (block[i * 4 + 3]);
234 }
235
236 // Extend the 16 words into 64 words
237 for (int i = 16; i < 64; i++) {
238 uint32_t s0 = rightRotate(w[i - 15], 7) ^ rightRotate(w[i - 15], 18) ^ (w[i - 15] >> 3);
239 uint32_t s1 = rightRotate(w[i - 2], 17) ^ rightRotate(w[i - 2], 19) ^ (w[i - 2] >> 10);
240 w[i] = w[i - 16] + s0 + w[i - 7] + s1;
241 }
242
243 // Initialize the working variables with the current hash values
244 uint32_t a = state[0];
245 uint32_t b = state[1];
246 uint32_t c = state[2];
247 uint32_t d = state[3];
248 uint32_t e = state[4];
249 uint32_t f = state[5];
250 uint32_t g = state[6];
251 uint32_t h = state[7];
252
253 // Main loop
254 for (int i = 0; i < 64; i++) {
255 uint32_t s1 = rightRotate(e, 6) ^ rightRotate(e, 11) ^ rightRotate(e, 25);
256 uint32_t ch = (e & f) ^ (~e & g);
257 uint32_t temp1 = h + s1 + ch + k[i] + w[i];
258 uint32_t s0 = rightRotate(a, 2) ^ rightRotate(a, 13) ^ rightRotate(a, 22);
259 uint32_t maj = (a & b) ^ (a & c) ^ (b & c);
260 uint32_t temp2 = s0 + maj;
261
262 h = g;
263 g = f;
264 f = e;
265 e = d + temp1;
266 d = c;
267 c = b;
268 b = a;
269 a = temp1 + temp2;
270 }
271
272 // Add the working variables to the current hash values
273 state[0] += a;
274 state[1] += b;
275 state[2] += c;
276 state[3] += d;
277 state[4] += e;
278 state[5] += f;
279 state[6] += g;
280 state[7] += h;
281 }
282
283 // Convert the final state (hash) to a hexadecimal string
284 std::string toHexString() {
285 std::stringstream ss;
286 for (int i = 0; i < 8; i++) {
287 ss << std::setw(8) << std::setfill('0') << std::hex << state[i];
288 }
289 return ss.str();
290 }
291
292 // Right rotate a 32-bit word
293 uint32_t rightRotate(uint32_t value, uint32_t count) {
294 return (value >> count) | (value << (32 - count));
295 }
296 };
