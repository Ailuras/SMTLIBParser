95 // compare two sorts
96 bool operator==(const Sort& other) const {
97 if((isInt() && other.isIntOrReal()) || (isIntOrReal() && other.isInt())){
98 return true;
99 }
100 else if((isReal() && other.isIntOrReal()) || (isIntOrReal() && other.isReal())){
101 return true;
102 }
103 else{
104 return kind == other.kind && name == other.name && arity == other.arity;
105 }
106 }
107 bool operator!=(const Sort& other) const {
108 return !(*this == other);
109 }

111 // print the sort
112 std::string toString() const {
113 switch (kind) {
114 case SORT_KIND::SK_BOOL: return "Bool";
115 case SORT_KIND::SK_INT: return "Int";
116 case SORT_KIND::SK_REAL: return "Real";
117 case SORT_KIND::SK_BV:
118 return "(_ BitVec " + std::to_string(children[0]->arity) + ")";
119 case SORT_KIND::SK_FP:
120 return "(_ FloatingPoint " + std::to_string(children[0]->arity) + " " + std::to_string(children[1]->arity) + ")";
121 case SORT_KIND::SK_STR: return "String";
122 case SORT_KIND::SK_ARRAY: return "(Array " + children[0]->toString() + " " + children[1]->toString() + ")";
123 case SORT_KIND::SK_DATATYPE: return "Datatype";
124 case SORT_KIND::SK_SET: return "Set";
125 case SORT_KIND::SK_RELATION: return "Relation";
126 case SORT_KIND::SK_BAG: return "Bag";
127 case SORT_KIND::SK_SEQ: return "Sequence";
128 case SORT_KIND::SK_UF: return "UF";
129 case SORT_KIND::SK_REG: return "RegLan";
130 case SORT_KIND::SK_EXT: return "ExtReal";
131 case SORT_KIND::SK_NAT: return "Natural";
132 case SORT_KIND::SK_RAND: return "Random";
133 case SORT_KIND::SK_INTOREAL: return "IntOrReal";
134 case SORT_KIND::SK_ALGEBRAIC: return "Algebraic";
135 case SORT_KIND::SK_TRANSCENDENTAL: return "Transcendental";
136 default: return "Unknown";
137 }
